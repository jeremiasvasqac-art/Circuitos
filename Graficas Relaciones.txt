Relacion Voltaje y Beta
import matplotlib.pyplot as plt

# Datos
beta = [50, 100, 150, 200]
voltaje_microvolts = [9.998, 4.992, 3.328, 2.496]

# Crear la gráfica
plt.figure(figsize=(8, 5))
plt.plot(beta, voltaje_microvolts, marker='o', linestyle='-', color='green', label='Voltaje vs Beta')
plt.title('Relación entre Voltaje y Beta del Transistor')
plt.xlabel('Beta (β)')
plt.ylabel('Voltaje (μV)')
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()



Relacion Corriente y beta
import matplotlib.pyplot as plt

# Datos
beta = [50, 100, 150, 200]
corriente_mA = [2, 2, 2, 2]  # Corriente constante de 2 mA

# Crear la gráfica
plt.figure(figsize=(8, 5))
plt.plot(beta, corriente_mA, marker='o', linestyle='-', color='red', label='Corriente vs Beta')
plt.title('Relación entre Corriente y Beta del Transistor')
plt.xlabel('Beta (β)')
plt.ylabel('Corriente (mA)')
plt.ylim(0, 3)  # Escala ajustada para mejor visualización
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()



Relacion Corriente vs Voltaje de la fuente

# Datos

Relacion Voltaje en una seccion vs Voltaje de la fuente
import matplotlib.pyplot as plt

# Datos
vcc = [10, 15, 20, 25]  # Voltaje de la fuente en V
voltaje_seccion = [1.136, 1.703, 2.269, 2.836]  # Voltaje en la sección en microvoltios

# Crear la gráfica
plt.figure(figsize=(8, 5))

plt.ylabel('Voltaje en la Sección (μV)')
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()
